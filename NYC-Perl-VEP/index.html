Step 5. Put in Code


sub strip_crnl {
    $_ = shift; s/$nl//g; s/$cr//g;
    return $_;
}

eval q{
    use Inline C  => <<'END_OF_C_CODE';
    char* strip_crnl(char* str) {
        char *s;
        char *s2 = str;
        for (s = str; *s; *s++) {
	    if (*s != '\n' && *s != '\r') {
              *s2++ = *s;
	    }
        }
        *s2 = '\0';
        return str;
    }
END_OF_C_CODE
